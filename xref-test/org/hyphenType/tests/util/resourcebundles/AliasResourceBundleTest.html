31      @Test
32      public void test1() {
33  	Map<String, String> map = new HashMap<String, String>();
34  	map.put("alias.x", "blablablablablabla");
35  	map.put("alias.y", "some.quite.long.and.troublesome.to.write.string");
36  	map.put("${x}", "123");
37  	map.put("${x}.text", "456");
38  	map.put("${y}", "789");
39  	map.put("${y}.x", "876");
40  	ConfigurableListResourceBundle clrb = new ConfigurableListResourceBundle(map);
41  	ResourceBundle rb = AliasResourceBundle.convert(clrb);
42  	System.out.println(rb.keySet());
43  	assertTrue(rb.containsKey("blablablablablabla"));
44  	assertTrue(rb.containsKey("blablablablablabla.text"));
45  	assertTrue(rb.containsKey("some.quite.long.and.troublesome.to.write.string"));
46  	assertTrue(rb.containsKey("some.quite.long.and.troublesome.to.write.string.x"));
47  	assertFalse(rb.containsKey("alias.x"));
48  	assertFalse(rb.containsKey("alias.y"));
49  	assertEquals("123", rb.getString("blablablablablabla"));
50  	assertEquals("456", rb.getString("blablablablablabla.text"));
51  	assertEquals("789", rb.getString("some.quite.long.and.troublesome.to.write.string"));
52  	assertEquals("876", rb.getString("some.quite.long.and.troublesome.to.write.string.x"));
53      }
