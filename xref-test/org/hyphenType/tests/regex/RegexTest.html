106     public void test5() throws Exception {

108 		RegexInterface1.class);
109 	RegexInterface1 ri1 = oe.options("@@bla", "@@xyz", "@x", "@02", "@77",
110 		"04"); // 04 will finish the array, since it does not match the
111 	// regex of value3
112 	assertEquals("@@bla", ri1.value());
113 	assertEquals("@@xyz", ri1.value2());

115 	assertArrayEquals(new String[] { "@02", "@77" }, ri1.value3());
116 	ArrayList<LexToken> expected = new ArrayList<LexToken>();
