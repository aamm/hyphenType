34  /**
35   * The only difference between this test and {@link SimplestTest} is that this
36   * one illustrates a unit test that utilizes {@link UnitTestingOptionExtractor}
37   * instead of {@link UnitTestingAppEngine}. Users should know how to utilize
38   * {@link UnitTestingOptionExtractor} when they are not testing applications.
39   * This is the case when the class that receives options object is not the main
40   * class in a stand alone execution. An example is classes under test receive
41   * interactive inputs from a console.
42   * 
43   * @author Aurelio Akira M. Matsui
44   */
45  public class GenericTest {
46  
47      @Test
48      public void test() throws InvalidOptionsInterfaceException, StringParsingErrorException, MandatoryValueNotFoundException, RegexMismatchException, StringParseException, MandatoryMapValueNotFoundException, MandatorySimpleArgumentNotFoundException, OptionsExtractorException {
49  
50          UnitTestingOptionExtractor<UserMadeOptions> extractor = new UnitTestingOptionExtractor<UserMadeOptions>(UserMadeOptions.class);
51          try {
52              UserMadeApplication.main(extractor.options("-x"));
53              fail("Should never reach this line since the line above should have thrown an exception.");
54          } catch (NonExceptionalExit e) {
55              assertTrue(e.exitIntCalled());
56              assertEquals(13, e.getStatusCodeInt());
57              assertEquals(UserMadeApplication.class.getName(), e.exitCallPoint().getClassName());
58              // Line number detection won't work unless classes were compiled
59              // using debug mode.
60              assertEquals(23, e.exitCallPoint().getLineNumber());
61              assertEquals("main", e.exitCallPoint().getMethodName());
62              assertEquals(11, UserMadeApplication.number);
63          }
64      }
